# Calculator Code File

def input_agent():
    """
    Asks the user for a number of terms and validates the input.
    
    Returns:
    int: A positive integer representing the number of terms.
    """
    while True:
        try:
            n = int(input("Enter a positive number of terms: "))
            if n > 0:
                return n
            else:
                print("Error: Input must be a positive integer.")
        except ValueError:
            print("Error: Input must be an integer.")


def fibonacci_agent(n):
    """
    Generates the Fibonacci series up to the nth term.
    
    Args:
    int: The number of terms in the series.
    
    Returns:
    list: A list of integers representing the Fibonacci series.
    """
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]
    else:
        fib_series = [0, 1]
        while len(fib_series) < n:
            fib_series.append(fib_series[-1] + fib_series[-2])
        return fib_series


def print_agent(result):
    """
    Prints the Fibonacci series to the console.
    
    Args:
    list: The Fibonacci series to be printed.
    """
    print("Fibonacci Series:")
    for num in result:
        print(num)


def main():
    n = input_agent()
    result = fibonacci_agent(n)
    print_agent(result)


if __name__ == "__main__":
    main()
