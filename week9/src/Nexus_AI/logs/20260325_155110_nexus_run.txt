import pandas as pd
import numpy as np

# Define a function to create and analyze a sample DataFrame
def analyze_data():
    # Create a dictionary with example data
    data = {'Name': ['John', 'Anna', 'Peter', 'Linda'],
            'Age': [28, 24, 35, 32],
            'Country': ['USA', 'UK', 'Australia', 'Germany']}

    # Create a DataFrame
    df = pd.DataFrame(data)

    # Populate the DataFrame with random scores
    np.random.seed(0)
    df['Score'] = np.random.randint(1, 100, 4)

    # Save the DataFrame as a CSV file
    try:
        df.to_csv('output.csv', index=False)
        print("Data has been saved to output.csv")
    except Exception as e:
        print(f"Error saving data: {e}")

    # Reload the saved DataFrame from the CSV file
    try:
        loaded_df = pd.read_csv('output.csv')
    except FileNotFoundError:
        print("Error reading data: File not found.")
        return

    # Showcase pandas analysis capabilities
    print(df.head())  # prints first 5 rows
    print(df.info())  # prints a concise summary of the DataFrame
    print(df.describe())  # prints summary statistics by column
    print(df.mean())  # prints the mean of the DataFrame

    # Append a summary of findings to the CSV file
    try:
        summary = loaded_df.describe()
        summary.to_csv('output.csv', mode='a', header=False, index=True)
        print("Summary of findings appended to output.csv")
    except Exception as e:
        print(f"Error appending summary: {e}")

# Execute the function
analyze_data()
