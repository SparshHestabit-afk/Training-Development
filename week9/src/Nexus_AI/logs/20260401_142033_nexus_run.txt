def generate_fibonacci_series(n):
    """
    Generates a Fibonacci series up to the nth term.

    Args:
        n (int): The number of terms in the series.

    Returns:
        list: A list of integers representing the Fibonacci series.
    """
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]

    fibonacci_series = [0, 1]
    while len(fibonacci_series) < n:

```python
# Arithmetic Calculator

# Constants
ERROR_INVALID_INPUT = "Invalid input! Please enter a number."
ERROR_DIVISION_BY_ZERO = "Error: Division by zero!"

def get_number(prompt):
    """
    Get a number from the user.

    Args:
        prompt (str): The prompt to display to the user.

    Returns:
        float: The number entered by the user.
    """
    while True:
        try:
            return float(input(prompt))
        except ValueError:
            print(ERROR_INVALID_INPUT)

def get_operator():
    """
    Get an operator from the user.

    Returns:
        str: The operator entered by the user.
    """
    operators = {
        '+': lambda x, y: x + y,
        '-': lambda x, y: x - y,
        '*': lambda x, y: x * y,
        '/': lambda x, y: x / y if y != 0 else float('inf')
    }

    while True:
        print("Choose an operator (+, -, *, /):")
        op = input().strip()
        if op in operators:
            return op, operators[op]
        else:
            print("Invalid operator. Please choose a valid operator.")

def calculate(num1, op, num2):
    """
    Calculate the result of the arithmetic operation.

    Args:
        num1 (float): The first number.
        op (str): The operator.
        num2 (float): The second number.

    Returns:
        float: The result of the operation.
    """
    return op(num1, num2)

def main():
    try:
        while True:
            num1 = get_number("Enter the first number: ")
            op, func = get_operator()
            num2 = get_number("Enter the second number: ")
            result = calculate(num1, func, num2)
            print(f"Result: {num1} {op} {num2} = {result}")
            cont = input("Do you want to continue? (y/n): ").strip().lower()
            if cont != 'y':
                break
    except ZeroDivisionError:
        print(ERROR_DIVISION_BY_ZERO)
    except Exception as e:
        print(f"An error occurred: {str(e)}")

if __name__ == "__main__":
    main()
```
