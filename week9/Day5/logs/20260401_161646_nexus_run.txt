import pandas as pd

# Load the CSV file
df = pd.read_csv('car_sales.csv')

# Insight 1: Total sales quantity and total sales revenue of car models in each region
total_sales_quantity = df.groupby('Region')['Sales Quantity'].sum()
total_sales_revenue = df.groupby('Region')['Sales Price'].apply(lambda x: x * x['Sales Quantity'].mean()).mean()

# Insight 2: Top-selling car models in each region based on sales quantity
top_models = df.groupby(['Region', 'Car Model'])['Sales Quantity'].sum().sort_values(ascending=False)

# Insight 3: Average sales price of car models in each region
avg_sales_price = df.groupby('Region')['Sales Price'].mean()

# Insight 4: Total sales revenue of car models in the North region
north_region_revenue = df[df['Region'] == 'North']['Sales Price'].apply(lambda x: x * df[df['Region'] == 'North']['Sales Quantity']).sum()

# Insight 5: Sales quantity distribution across different regions
sales_quantity_distribution = df['Region'].value_counts()

print("Insight 1: Total sales quantity and total sales revenue of car models in each region")
print(total_sales_quantity)
print(total_sales_revenue)

print("\nInsight 2: Top-selling car models in each region based on sales quantity")
print(top_models)
